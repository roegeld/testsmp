\begin{mplibcode}
  beginfig(10);
  numeric u;u=1cm;
  numeric r;
  pair A,B,C,D,E,F,I;
  A=origin;
  B-A=(7u,u);C-A=(2u,4u);
  draw A--B--C--cycle;
  label.llft(btex $A$ etex,A);
  label.rt(btex $B$ etex,B);
  label.top(btex $C$ etex,C);
  D=.5[A,B];E=.5[B,C];F=.5[C,A];
  I=whatever[D,D+((B-A) rotated 90)]=whatever[E,E+((C-B) rotated 90)];
  r=arclength(A--I);
  draw fullcircle scaled 2r shifted I withcolor red;
  endfig;
